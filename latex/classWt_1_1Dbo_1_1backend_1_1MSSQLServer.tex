Throws an exception if there was a problem, otherwise returns true.

\DoxyCodeLine{Driver=\{ODBC Driver 13 \textcolor{keywordflow}{for} SQL Server\};}
\DoxyCodeLine{Server=localhost;}
\DoxyCodeLine{UID=SA;}
\DoxyCodeLine{PWD=\{example password\};}
\DoxyCodeLine{\mbox{\hyperlink{classDatabase}{Database}}=example\_db;}
\end{DoxyCode}

163 template <class C>
164 void collection<C>::iterator::shared_impl::fetchNextRow()
165 {
166  if (ended_)
167  throw Exception("set< ptr<C> >::operator++ : beyond end.");
168  else if (queryEnded_) {
169  posPastQuery_++;
170  if (posPastQuery_ == collection_.manualModeInsertions().size())
171  ended_ = true;
172  else
173  current_ = collection_.manualModeInsertions()[posPastQuery_];
174  return;
175  }
176 
177  if (!statement_ || !statement_->nextRow()) {
178  queryEnded_ = true;
179  if (collection_.manualModeInsertions().size() == 0)
180  ended_ = true;
181  if (statement_) {
182  statement_->done();
183  collection_.iterateDone();
184  }
185  } else {
186  int column = 0;
187  current_
188  = query_result_traits<C>::load(*collection_.session(), *statement_, column);
189 
190  Impl::Helper<C>::skipIfRemoved(*this);
191  }
192 }

488 template <class C>
489 typename collection<C>::size_type collection<C>::size() const
490 {
491  if (type_ == QueryCollection && data_.query->size != -1)
492  return data_.query->size;
493 
494  SqlStatement *countStatement = nullptr;
495 
496  if (session_ && session_->flushMode() == FlushMode::Auto)
497  session_->flush();
498 
499  if (type_ == QueryCollection)
500  countStatement = data_.query->countStatement;
501  else {
502  if (data_.relation.sql) {
503  const std::string *sql = data_.relation.sql;
504  std::size_t f = Impl::ifind(*sql, " from ");
505  std::string countSql = "select count(1)" + sql->substr(f);
506 
507  countStatement = session_->getOrPrepareStatement(countSql);
508  int column = 0;
509  data_.relation.dbo->bindId(countStatement, column);
510  }
511  }
512 
513  if (countStatement) {
514  ScopedStatementUse use(countStatement);
515 
516  countStatement->execute();
517 
518  if (!countStatement->nextRow())
519  throw Exception("collection<C>::size(): no result?");
520 
521  int result;
522  if (!countStatement->getResult(0, &result))
523  throw Exception("collection<C>::size(): null?");
524 
525  if (countStatement->nextRow())
526  throw Exception("collection<C>::size(): multiple results?");
527 
528  if (type_ == QueryCollection) {
529  data_.query->size = result;
530  data_.query->countStatement = nullptr;
531  }
532 
533  if (type_ != QueryCollection) {
534  result += static_cast<int>(manualModeInsertions_.size());
535  result -= static_cast<int>(manualModeRemovals_.size());
536  }
537 
538  return static_cast<typename collection<C>::size_type>(result);
539  } else
540  return 0;
541 }

543 template <class C>
544 Query<C, DynamicBinding> collection<C>::find() const
545 {
546  if (type_ != RelationCollection)
547  throw Exception("collection<C>::find() "
548  "only for a many-side relation collection.");
549 
550  if (session_ && data_.relation.sql) {
551  const std::string *sql = data_.relation.sql;
552  std::size_t f = Impl::ifind(*sql, " from ");
553  std::size_t w = Impl::ifind(*sql, " where ");
554  std::string tableName = sql->substr(f + 6, w - f - 6);
555 
556  Query<C, DynamicBinding> result = Query<C, DynamicBinding>
557  (*session_, tableName, "").where(sql->substr(w + 7));
558 
559  if (!data_.relation.dbo->isPersisted())
560  data_.relation.dbo->flush();
561  data_.relation.dbo->bindId(result.parameters_);
562 
563  return result;
564  } else
565  return Query<C, DynamicBinding>();

568 template <class C>
569 void collection<C>::insert(C c)
570 {
571  RelationData& relation = data_.relation;
572 
573  if (type_ != RelationCollection || relation.setInfo == nullptr)
574  throw Exception("collection<C>::insert() only for a relational "
575  "collection.");
576 
577  if (session_->flushMode() == FlushMode::Auto) {
578  if (relation.dbo) {
579  relation.dbo->setDirty();
580  if (relation.dbo->session())
581  relation.dbo->session()->add(c);
582  }
583  } else if (session_->flushMode() == FlushMode::Manual) {
584  manualModeInsertions_.push_back(c);
585  }
586 
587  if (relation.setInfo->type == ManyToMany) {
588  if (!relation.activity)
589  relation.activity = new Activity();
590 
591  bool wasJustErased = relation.activity->erased.erase(c) > 0;
592  relation.activity->transactionErased.erase(c);
593 
594  if (!wasJustErased && !relation.activity->transactionInserted.count(c))
595  relation.activity->inserted.insert(c);
596  } else {
597  SetReciproceAction setPtr(session_, relation.setInfo->joinName,
598  relation.dbo);
599  setPtr.visit(*c.modify());
600  }
601 }

603 template <class C>
604 void collection<C>::erase(C c)
605 {
606  RelationData& relation = data_.relation;
607 
608  if (type_ != RelationCollection || relation.setInfo == nullptr)
609  throw Exception("collection<C>::erase() only for a relational relation.");
610 
611  if (relation.dbo)
612  relation.dbo->setDirty();
613 
614  if (relation.setInfo->type == ManyToMany) {
615  if (!relation.activity)
616  relation.activity = new Activity();
617 
618  bool wasJustInserted = relation.activity->inserted.erase(c) > 0;
619  relation.activity->transactionInserted.erase(c);
620 
621  if (!wasJustInserted && !relation.activity->transactionErased.count(c))
622  relation.activity->erased.insert(c);
623  } else {
624  SetReciproceAction setPtr(session_, relation.setInfo->joinName, nullptr);
625  setPtr.visit(*c.modify());
626  }
627 
628  typename std::vector<C>::iterator pos =
629  std::find(manualModeInsertions_.begin(), manualModeInsertions_.end(), c);
630  if (pos != manualModeInsertions_.end())
631  manualModeInsertions_.erase(pos);
632 
633  if (session_->flushMode() == FlushMode::Manual) {
634  manualModeRemovals_.push_back(c);
635  }

638 template <class C>
639 void collection<C>::clear()
640 {
641  RelationData& relation = data_.relation;
642 
643  if (type_ != RelationCollection || relation.setInfo == nullptr)
644  throw Exception("collection<C>::clear() only for a relational relation.");
645 
646  if (relation.setInfo->type == ManyToMany) {
647  if (relation.activity) {
648  relation.activity->transactionInserted.clear();
649  relation.activity->transactionErased.clear();
650  }
651  }
652 
653  if (relation.dbo) {
654  const std::string *sql = relation.sql;
655  std::string deleteSql;
656 
657  if (relation.setInfo->type == ManyToMany) {
658  std::size_t o = Impl::ifind(*sql, " on ");
659  std::size_t j = Impl::ifind(*sql, " join ");
660  std::size_t w = Impl::ifind(*sql, " where ");
661  deleteSql = "delete from " + sql->substr(j + 5, o - j - 5)
662  + sql->substr(w);
663  } else {
664  std::size_t f = Impl::ifind(*sql, " from ");
665  deleteSql = "delete" + sql->substr(f);
666  }
667 
668  Call call = session_->execute(deleteSql);
669  int column = 0;
670  relation.dbo->bindId(call.statement_, column);
671  call.run();
672  }
673  manualModeInsertions_.clear();
674  manualModeRemovals_.clear();

677 template <class C>
678 int collection<C>::count(C c) const
679 {
680  if (!session_)
681  throw Exception("collection<C>::count() only for a collection "
682  "that is bound to a session.");
683 
684  if (session_->flushMode() == FlushMode::Auto)
685  session_->flush();
686 
687  if (type_ != RelationCollection)
688  throw Exception("collection<C>::count() only for a relational "
689  "relation.");
690 
691  if (!c)
692  return 0;
693 
694  const RelationData& relation = data_.relation;
695  Impl::MappingInfo *mapping
696  = session_->getMapping(relation.setInfo->tableName);
697 
698  Query<C, DynamicBinding> q = find().where(mapping->idCondition);
699  c.obj()->bindId(q.parameters_);
700  std::size_t result = q.resultList().size();
701 
702  result +=
703  std::count(manualModeInsertions_.begin(), manualModeInsertions_.end(), c);
704  result -=
705  std::count(manualModeRemovals_.begin(), manualModeRemovals_.end(), c);
706  return static_cast<int>(result);

364 template<typename V>
365 void SaveBaseAction::act(const FieldRef<V>& field)
366 {
367  if (auxIdOnly_ && !(field.flags() & FieldRef<V>::AuxId))
368  return;
369 
370  if (pass_ == Self) {
371  if (bindNull_)
372  statement_->bindNull(column_++);
373  else
374  field.bindValue(statement_, column_++);
375  }
376 }

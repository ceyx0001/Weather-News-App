84 void observing_ptr<T>::doMove(observing_ptr<S>& other) noexcept
85 {
86  if (&impl_ == &other.impl_)
87  return;

89  if (impl_.observed_) {

92  impl_.observed_->removeObserver(&impl_);
93  }

96  other.impl_ = Impl::observing_ptr_base{};
97  if (impl_.observed_)

99 }

456  static void renderiv(std::ostream &os,
457  const IntBuffer& a,
458  WGLWidget::GLenum type)
459  {
460  switch (type) {
461  case WGLWidget::BYTE:
462  os << "new Int8Array([";
463  break;
464  case WGLWidget::UNSIGNED_BYTE:
465  os << "new Uint8Array([";
466  break;
467  case WGLWidget::SHORT:
468  os << "new Int16Array([";
469  break;
470  case WGLWidget::UNSIGNED_SHORT:
471  os << "new Uint16Array([";
472  break;
473  case WGLWidget::INT:
474  os << "new Int32Array([";
475  break;
476  default:
477  // Should we warn?
478  case WGLWidget::UNSIGNED_INT:
479  os << "new Uint32Array([";
480  break;
481  }
482  char buf[30];
483  unsigned i;
484  for (i = 0; i < a.size(); i++)
485  os << (i == 0 ? "" : ",") << makeInt(a[i], buf);
486  os << "])";

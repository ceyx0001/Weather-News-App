279 template <class Result>
280 void QueryModel<Result>::cacheRow(int row) const
281 {
282  if (row < cacheStart_
283  || row >= cacheStart_ + static_cast<int>(cache_.size())) {
284  cacheStart_ = std::max(row - batchSize_ / 4, 0);
285 
286  int qOffset = cacheStart_;
287  if (queryOffset_ > 0)
288  qOffset += queryOffset_;
289  query_.offset(qOffset);
290 
291  int qLimit = batchSize_;
292  if (queryLimit_ > 0)
293  qLimit = std::min(batchSize_, queryLimit_ - cacheStart_);
294  query_.limit(qLimit);
295 
296  Transaction transaction(query_.session());
297 
298  collection<Result> results = query_.resultList();
299  cache_.clear();
300  cache_.insert(cache_.end(), results.begin(), results.end());
301 
302  for (unsigned i = 0; i < cache_.size(); ++i) {
303  long long id = resultId(cache_[i]);
304  if (id != -1)
305  stableIds_[cacheStart_ + i] = id;
306  }
307  if (static_cast<int>(cache_.size()) < qLimit
308  && qOffset == 0 && cachedRowCount_ == -1)
309  cachedRowCount_ = cache_.size();
310 
311  transaction.commit();
312  }
313 }

402 template <class Result>
403 bool QueryModel<Result>::insertRows(int row, int count,
404  const WModelIndex& parent)
405 {
406  if (row != rowCount())
407  throw Exception("QueryModel: only supporting row insertion at end");
408 
409  beginInsertRows(parent, row, row + count - 1);
410 
411  for (int i = 0; i < count; ++i) {
412  Result r = createRow();
413  addRow(r);
414 
415  /*
416  * Insert also into cache, this avoids a useless insert+query
417  * when insertion is followed by setData() calls.
418  */
419  if (cacheStart_ != -1
420  && cacheStart_ + (int)cache_.size() == row + i)
421  cache_.push_back(r);
422  }
423 
424  cachedRowCount_ += count;
425 
426  endInsertRows();
427 
428  return true;
429 }

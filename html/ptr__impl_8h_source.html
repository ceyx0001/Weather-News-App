63 template <class C>
64 void MetaDbo<C>::flush()
65 {
66  checkNotOrphaned();
67 
68  if (state_ & NeedsDelete) {
69  state_ &= ~NeedsDelete;
70 
71  try {
72  session()->implDelete(*this);
73  setTransactionState(DeletedInTransaction);
74  } catch (...) {
75  setTransactionState(DeletedInTransaction);
76  throw;
77  }
78  } else if (state_ & NeedsSave) {
79  state_ &= ~NeedsSave;
80  state_ |= Saving;
81 
82  try {
83  session()->implSave(*this);
84  setTransactionState(SavedInTransaction);
85  } catch (...) {
86  setTransactionState(SavedInTransaction);
87  throw;
88  }
89  } else if (state_ & Saving) {
90  /*
91  * This must be because of a circular relational dependency:
92  * A belongsTo(B)
93  * B belongsTo(A)
94  *
95  * Could also be because of A belongsTo(A) which could be perfectly
96  * if the ptr is assigned after the object itself has been saved first.
97  */
98  // throw Exception("Wt::Dbo::ptr::flush(): circular dependency detected!");
99  }
100 }

146 template <class C>
147 void MetaDbo<C>::doTransactionDone(bool success)
148 {
149  Session *s = session();
150 
151  if (success) {
152  if (deletedInTransaction()) {
153  prune();
154  setSession(nullptr);
155  } else if (savedInTransaction()) {
156  setVersion(version() + 1);
157  setState(Persisted);
158  }
159  } else {
160  /*
161  * This undoing of the transaction isn't consistent with objects that
162  * haven't been flushed.
163  */
164  if (deletedInTransaction()) {
165  state_ |= NeedsDelete;
166  session()->needsFlush(this);
167  } else if (savedInTransaction()) {
168  if (isNew()) {
169  prune();
170  } else {
171  /*
172  * If we support changing the Id, then we need to restore the old
173  * Id here.
174  */
175  state_ |= NeedsSave;
176  session()->needsFlush(this);
177  }
178  }
179  }
180 
181  if (obj_)
182  s->implTransactionDone(*this, success);
183 
184  resetTransactionState();
185 }

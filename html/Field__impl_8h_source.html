125 template <class C>
126 template <class A>
127 void PtrRef<C>::visit(A& action, Session *session) const
128 {
129  typename dbo_traits<C>::IdType id;
130 
131  if (action.setsValue())
132  id = dbo_traits<C>::invalidId();
133  else
134  id = value_.id();
135 
136  std::string idFieldName = "stub";
137  int size = -1;
138 
139  if (session) {
140  Impl::MappingInfo *mapping = session->getMapping<C>();
141  action.actMapping(mapping);
142  idFieldName = mapping->naturalIdFieldName;
143  size = mapping->naturalIdFieldSize;
144 
145  if (idFieldName.empty())
146  idFieldName = mapping->surrogateIdFieldName;
147  }
148 
149  if (literalForeignKey()) {
150  field(action, id, name_, size);
151  } else {
152  field(action, id, name_ + "_" + idFieldName, size);
153  }
154 
155  LoadLazyHelper<C, A>::loadLazy(value_, id, session);
156 }
